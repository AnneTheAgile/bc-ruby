def cartAsMapOfProductsAndQuantities
  aMap = @products.sort.each_with_object(Hash.new) {|i, m| m[i]+=1 rescue m[i]=1}
  #print(["-cartAsMap=",aMap.inspect,"\n"].join)
  aMap
end
